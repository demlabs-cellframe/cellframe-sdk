160 static dap_binary_tree_t *s_tree_insert(dap_binary_tree_t *a_elm, dap_binary_tree_key_t a_key, void *a_data)
161 {
162  if (a_elm == NULL) {
163  dap_binary_tree_t* l_elm = DAP_NEW_Z(dap_binary_tree_t);
164  l_elm->left = l_elm->right = NULL;
165  l_elm->key = a_key;
166  l_elm->data = a_data;
167  return l_elm;
168  }
169  if (KEY_LS(a_key, a_elm->key)) {
170  a_elm->left = s_tree_insert(a_elm->left, a_key, a_data);
171  } else if (KEY_GT(a_key, a_elm->key)) {
172  a_elm->right = s_tree_insert(a_elm->right, a_key, a_data);
173  } else { //if KEY_EQ(a_key, a_elm->key)
174  a_elm->data = a_data;
175  }
176  return a_elm;
177 }

199 static dap_binary_tree_t *s_tree_delete(dap_binary_tree_t *a_elm, dap_binary_tree_key_t a_key)
200 {
201  if (a_elm == NULL) {
202  return a_elm;
203  }
204  if (KEY_LS(a_key, a_elm->key)) {
205  a_elm->left = s_tree_delete(a_elm->left, a_key);
206  } else if (KEY_GT(a_key, a_elm->key)) {
207  a_elm->right = s_tree_delete(a_elm->right, a_key);
208  } else if (a_elm->left && a_elm->right) {
209  dap_binary_tree_t *l_tmp = s_tree_minimum(a_elm->right);
210  a_elm->key = l_tmp->key;
211  a_elm->data = l_tmp->data;
212  a_elm->right = s_tree_delete(a_elm->right, a_elm->key);
213  } else if (a_elm->left) {
214  dap_binary_tree_t * l_elm_old_left = a_elm->left;
215  DAP_DELETE(a_elm->data);
216  DAP_DELETE(a_elm);
217  a_elm = l_elm_old_left;
218  } else if (a_elm->right) {
219  dap_binary_tree_t * l_elm_old_right = a_elm->right;
220  DAP_DELETE(a_elm->data);
221  DAP_DELETE(a_elm);
222  a_elm = l_elm_old_right;
223  } else {
224  DAP_DELETE(a_elm->data);
225  DAP_DELETE(a_elm);
226  a_elm = NULL;
227  }
228  return a_elm;
229 }
