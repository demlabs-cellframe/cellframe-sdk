319 static void *s_log_thread_proc(void *arg) {
320  (void) arg;
321  for ( ; !s_log_term_signal; ) {
322  pthread_mutex_lock(&s_log_mutex);
323  for ( ; s_log_count == 0 && !s_log_term_signal; ) {
324  pthread_cond_wait(&s_log_cond, &s_log_mutex);
325  }
326  if (s_log_count) {
327  log_str_t *elem, *tmp;
328  if(s_log_file) {
329  if(!dap_file_test(s_log_file_path)) {
330  fclose(s_log_file);
331  s_log_file = fopen(s_log_file_path, "a");
332  if( s_log_file == NULL) {
333  dap_mkdir_with_parents(s_log_dir_path);
334  s_log_file = fopen( s_log_file_path , "w" );
335  }
336  }
337  }
338  DL_FOREACH_SAFE(s_log_buffer, elem, tmp) {
339  if(s_log_file)
340  fwrite(elem->str + elem->offset, strlen(elem->str) - elem->offset, 1, s_log_file);
341  fwrite(elem->str, strlen(elem->str), 1, stdout);
342 
343  DL_DELETE(s_log_buffer, elem);
344  DAP_FREE(elem);
345  --s_log_count;
346 
347  if(s_log_file)
348  fflush(s_log_file);
349  fflush(stdout);
350  }
351  }
352  pthread_mutex_unlock(&s_log_mutex);
353  }
354  return NULL;
355 }

364 void _log_it(const char *a_log_tag, enum dap_log_level a_ll, const char *a_fmt, ...) {
365  if ( a_ll < s_dap_log_level || a_ll >= 16 || !a_log_tag )
366  return;
367  log_str_t *l_log_string = DAP_NEW_Z(log_str_t);
368  size_t offset2 = sizeof(l_log_string->str) - 2;
369  strncpy(l_log_string->str, s_ansi_seq_color[a_ll], offset2);
370  l_log_string->offset = s_ansi_seq_color_len[a_ll];
371  s_update_log_time(l_log_string->str + l_log_string->offset);
372  size_t offset = strlen(l_log_string->str);
373  offset += dap_snprintf(l_log_string->str + offset, offset2, "%s[%s%s", s_log_level_tag[a_ll], a_log_tag, "] ");
374  offset2 -= offset;
375  va_list va;
376  va_start( va, a_fmt );
377  size_t l_offset = dap_vsnprintf(l_log_string->str + offset, offset2, a_fmt, va);
378  offset = (l_offset < offset2) ? offset + l_offset : offset;
379  offset2 = (l_offset < offset2) ? offset2 - offset : 0;
380  va_end( va );
381  char *dummy = (offset2 == 0) ? memcpy(&l_log_string->str[sizeof(l_log_string->str) - 6], "...\n\0", 5)
382  : memcpy(&l_log_string->str[offset], "\n", 1);
383  UNUSED(dummy);
384  pthread_mutex_lock(&s_log_mutex);
385  DL_APPEND(s_log_buffer, l_log_string);
386  ++s_log_count;
387  pthread_cond_signal(&s_log_cond);
388  pthread_mutex_unlock(&s_log_mutex);
389 }

490 int timespec_diff(struct timespec *a_start, struct timespec *a_stop, struct timespec *a_result)
491 {
492  if(!a_start || !a_stop)
493  return 0;
494  if(!a_result) {
495  struct timespec l_time_tmp = { 0 };
496  a_result = &l_time_tmp;

500  a_result->tv_nsec = a_stop->tv_nsec - a_start->tv_nsec + 1000000000;

504  }
505

778 size_t dap_hex2bin(uint8_t *a_out, const char *a_in, size_t a_len)

784  if (!a_in || !a_out)
785  return 0;
