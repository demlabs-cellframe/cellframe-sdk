159 DAP_STATIC_INLINE void *_dap_aligned_alloc( uintptr_t alignment, uintptr_t size )
160 {
161  uintptr_t ptr = (uintptr_t) DAP_MALLOC( size + (alignment * 2) + sizeof(void *) );
162 
163  if ( !ptr )
164  return (void *)ptr;
165 
166  uintptr_t al_ptr = ( ptr + sizeof(void *) + alignment) & ~(alignment - 1 );
167  ((uintptr_t *)al_ptr)[-1] = ptr;
168 
169  return (void *)al_ptr;
170 }
171 
172 DAP_STATIC_INLINE void *_dap_aligned_realloc( uintptr_t alignment, void *bptr, uintptr_t size )
173 {
174  uintptr_t ptr = (uintptr_t) DAP_REALLOC( bptr, size + (alignment * 2) + sizeof(void *) );
175 
176  if ( !ptr )
177  return (void *)ptr;
178 
179  uintptr_t al_ptr = ( ptr + sizeof(void *) + alignment) & ~(alignment - 1 );
180  ((uintptr_t *)al_ptr)[-1] = ptr;
181 
182  return (void *)al_ptr;
183 }

349 {\
350  uintptr_t _len = len; \
351  uint16_t *__restrict _out = (uint16_t *__restrict)out; \
352  uint64_t *__restrict _in = (uint64_t *__restrict)in;\
353 \
354  while ( _len ) {\
355  uint64_t _val = *_in ++;\
356  _out[0] = htoa_lut256[ _val & 0x00000000000000FF ];\
357  _out[1] = htoa_lut256[ (_val & 0x000000000000FF00) >> 8 ];\
358  _out[2] = htoa_lut256[ (_val & 0x0000000000FF0000) >> 16 ];\
359  _out[3] = htoa_lut256[ (_val & 0x00000000FF000000) >> 24 ];\
360  _out[4] = htoa_lut256[ (_val & 0x000000FF00000000) >> 32 ];\
361  _out[5] = htoa_lut256[ (_val & 0x0000FF0000000000) >> 40 ];\
362  _out[6] = htoa_lut256[ (_val & 0x00FF000000000000) >> 48 ];\

364  _out += 8;\
365  _len -= 8;\
366  }\
