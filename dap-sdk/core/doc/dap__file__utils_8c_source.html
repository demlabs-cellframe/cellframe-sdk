134 int dap_mkdir_with_parents(const char *a_dir_path)
135 {
136  // validation of a pointer
137  if(a_dir_path == NULL || a_dir_path[0] == '\0') {
138  errno = EINVAL;
139  return -1;
140  }
141  char path[strlen(a_dir_path) + 1];
142  memset(path, '\0', strlen(a_dir_path) + 1);
143  memcpy(path, a_dir_path, strlen(a_dir_path));
144  char *p;
145  // skip the root component if it is present, i.e. the "/" in Unix or "C:\" in Windows
146 #ifdef _WIN32
147  if(((path[0] >= 'a' && path[0] <= 'z') || (path[0] >= 'A' && path[0] <= 'Z'))
148  && (path[1] == ':') && DAP_IS_DIR_SEPARATOR(path[2])) {
149  p = path + 3;
150  }
151 #else
152  if (DAP_IS_DIR_SEPARATOR(path[0])) {
153  p = path + 1;
154  }
155 #endif
156  else
157  p = path;
158 
159  do {
160  while(*p && !DAP_IS_DIR_SEPARATOR(*p))
161  p++;
162 
163  if(!*p)
164  p = NULL;
165  else
166  *p = '\0';
167 
168  if(!dap_dir_test(path)) {
169 #ifdef _WIN32
170  int result = mkdir(path);
171 #else
172  int result = mkdir(path, S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH | S_IWOTH);
173  chmod(path, S_IRWXU | S_IRWXG | S_IROTH | S_IXOTH | S_IWOTH);
174 #endif
175  if(result == -1) {
176  errno = ENOTDIR;
177  return -1;
178  }
179  }
180  if(p) {
181  *p++ = DAP_DIR_SEPARATOR;
182  while(*p && DAP_IS_DIR_SEPARATOR(*p))
183  p++;
184  }
185  } while(p);
186  return 0;
187 }

290 char* dap_path_get_dirname(const char *a_file_name)
291 {
292  log_it(L_DEBUG,"dap_path_get_dirname(a_file_name=\"%s\")", a_file_name);
293  char *l_base;
294  size_t l_len;
295 
296  dap_return_val_if_fail(a_file_name != NULL, NULL);
297 
298  l_base = strrchr(a_file_name, DAP_DIR_SEPARATOR);
299 
300 #ifdef _WIN32
301  {
302  char *l_q;
303  l_q = strrchr (a_file_name, '/');
304  if (l_base == NULL || (l_q != NULL && l_q > l_base))
305  l_base = l_q;
306  }
307 #endif
308 
309  if(!l_base)
310  {
311 #ifdef _WIN32
312  if (dap_ascii_isalpha(a_file_name[0]) && a_file_name[1] == ':')
313  {
314  char l_drive_colon_dot[4];
315 
316  l_drive_colon_dot[0] = a_file_name[0];
317  l_drive_colon_dot[1] = ':';
318  l_drive_colon_dot[2] = '.';
319  l_drive_colon_dot[3] = '\0';
320 
321  return dap_strdup (l_drive_colon_dot);
322  }
323 #endif
324  log_it(L_DEBUG,"l_base is NULL, return dup of .");
325  return dap_strdup(".");
326  }
327 
328  while(l_base > a_file_name && DAP_IS_DIR_SEPARATOR(*l_base))
329  l_base--;
330 
331 #ifdef _WIN32
332  /* base points to the char before the last slash.
333  *
334  * In case file_name is the root of a drive (X:\) or a child of the
335  * root of a drive (X:\foo), include the slash.
336  *
337  * In case file_name is the root share of an UNC path
338  * (\\server\share), add a slash, returning \\server\share\ .
339  *
340  * In case file_name is a direct child of a share in an UNC path
341  * (\\server\share\foo), include the slash after the share name,
342  * returning \\server\share\ .
343  */
344  if (l_base == a_file_name + 1 &&
345  dap_ascii_isalpha(a_file_name[0]) &&
346  a_file_name[1] == ':')
347  l_base++;
348  else if (DAP_IS_DIR_SEPARATOR (a_file_name[0]) &&
349  DAP_IS_DIR_SEPARATOR (a_file_name[1]) &&
350  a_file_name[2] &&
351  !DAP_IS_DIR_SEPARATOR (a_file_name[2]) &&
352  l_base >= a_file_name + 2)
353  {
354  const char *l_p = a_file_name + 2;
355  while (*l_p && !DAP_IS_DIR_SEPARATOR (*l_p))
356  l_p++;
357  if (l_p == l_base + 1)
358  {
359  l_len = (uint32_t) strlen (a_file_name) + 1;
360  l_base = DAP_NEW_SIZE (char, l_len + 1);
361  strcpy (l_base, a_file_name);
362  l_base[l_len-1] = DAP_DIR_SEPARATOR;
363  l_base[l_len] = 0;
364  return l_base;
365  }
366  if (DAP_IS_DIR_SEPARATOR (*l_p))
367  {
368  l_p++;
369  while (*l_p && !DAP_IS_DIR_SEPARATOR (*l_p))
370  l_p++;
371  if (l_p == l_base + 1)
372  l_base++;
373  }
374  }
375 #endif
376 
377  l_len = (uint32_t) 1 + l_base - a_file_name;
378  l_base = DAP_NEW_SIZE(char, l_len + 1);
379  memcpy(l_base, a_file_name, l_len);
380  l_base[l_len] = 0;
381  log_it(L_DEBUG,"l_base=%s",l_base);
382  return l_base;
383 }

449 static bool get_contents_stdio(const char *filename, FILE *f, char **contents, size_t *length)
450 {
451  char buf[4096];
452  size_t bytes; /* always <= sizeof(buf) */
453  char *str = NULL;
454  size_t total_bytes = 0;
455  size_t total_allocated = 0;
456  char *tmp;
457  assert(f != NULL);
458  while(!feof(f)) {
459  int save_errno;
460 
461  bytes = fread(buf, 1, sizeof(buf), f);
462  save_errno = errno;
463 
464  if(total_bytes > ULONG_MAX - bytes)
465  goto file_too_large;
466 
467  /* Possibility of overflow eliminated above. */
468  while(total_bytes + bytes >= total_allocated) {
469  if(str) {
470  if(total_allocated > ULONG_MAX / 2)
471  goto file_too_large;
472  total_allocated *= 2;
473  }
474  else {
475  total_allocated = MIN(bytes + 1, sizeof(buf));
476  }
477 
478  tmp = DAP_REALLOC(str, total_allocated);
479 
480  if(tmp == NULL)
481  goto error;
482 
483  str = tmp;
484  }
485 
486  if(ferror(f))
487  goto error;
488 
489  assert(str != NULL);
490  memcpy(str + total_bytes, buf, bytes);
491  total_bytes += bytes;
492  }
493 
494  fclose(f);
495 
496  if(total_allocated == 0)
497  {
498  str = DAP_NEW_SIZE(char, 1);
499  total_bytes = 0;
500  }
501 
502  str[total_bytes] = '\0';
503 
504  if(length)
505  *length = total_bytes;
506 
507  *contents = str;
508 
509  return true;
510 
511  file_too_large:
512  error:
513 
514  DAP_DELETE(str);
515  fclose(f);
516  return false;
517 }

522  size_t *length)
523 {

530 
531  alloc_size = size + 1;
532  buf = DAP_NEW_SIZE(char, alloc_size);
533 
534  if(buf == NULL) {
535  goto error;
536  }
537 
538  bytes_read = 0;
539  while(bytes_read < size) {
540  size_t rc;

543

546  DAP_DELETE(buf);

548  }
549  }

552  else
553  bytes_read += rc;
554  }
555 
556  buf[bytes_read] = '\0';
557 
558  if(length)
559  *length = bytes_read;
560 
561  *contents = buf;
562 
563  close(fd);
564 
565  return true;
566 
567  error:
568 
569  close(fd);
570 
571  return false;
