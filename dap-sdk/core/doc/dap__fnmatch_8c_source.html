228 /* We have to convert the wide character string in a multibyte string. But
229  we know that the character class names consist of alphanumeric characters
230  from the portable character set, and since the wide character encoding
231  for a member of the portable character set is the same code point as
232  its single-byte encoding, we can use a simplified method to convert the
233  string to a multibyte character string. */
234 static wctype_t
235 is_char_class (const wchar_t *wcs)
236 {
237  char s[CHAR_CLASS_MAX_LENGTH + 1];
238  char *cp = s;
239 
240  do
241  {
242  /* Test for a printable character from the portable character set. */
243 # ifdef _LIBC
244  if (*wcs < 0x20 || *wcs > 0x7e
245  || *wcs == 0x24 || *wcs == 0x40 || *wcs == 0x60)
246  return (wctype_t) 0;
247 # else
248  switch (*wcs)
249  {
250  case L' ': case L'!': case L'"': case L'#': case L'%':
251  case L'&': case L'\'': case L'(': case L')': case L'*':
252  case L'+': case L',': case L'-': case L'.': case L'/':
253  case L'0': case L'1': case L'2': case L'3': case L'4':
254  case L'5': case L'6': case L'7': case L'8': case L'9':
255  case L':': case L';': case L'<': case L'=': case L'>':
256  case L'?':
257  case L'A': case L'B': case L'C': case L'D': case L'E':
258  case L'F': case L'G': case L'H': case L'I': case L'J':
259  case L'K': case L'L': case L'M': case L'N': case L'O':
260  case L'P': case L'Q': case L'R': case L'S': case L'T':
261  case L'U': case L'V': case L'W': case L'X': case L'Y':
262  case L'Z':
263  case L'[': case L'\\': case L']': case L'^': case L'_':
264  case L'a': case L'b': case L'c': case L'd': case L'e':
265  case L'f': case L'g': case L'h': case L'i': case L'j':
266  case L'k': case L'l': case L'm': case L'n': case L'o':
267  case L'p': case L'q': case L'r': case L's': case L't':
268  case L'u': case L'v': case L'w': case L'x': case L'y':
269  case L'z': case L'{': case L'|': case L'}': case L'~':
270  break;
271  default:
272  return (wctype_t) 0;
273  }
274 # endif
275 
276  /* Avoid overrunning the buffer. */
277  if (cp == s + CHAR_CLASS_MAX_LENGTH)
278  return (wctype_t) 0;
279 
280  *cp++ = (char) *wcs++;
281  }
282  while (*wcs != L'\0');
283 
284  *cp = '\0';
285 
286 # ifdef _LIBC
287  return __wctype (s);
288 # else
289  return wctype (s);
290 # endif
291 }

298 int dap_fnmatch (const char *pattern, const char *string, int flags)
299 {
300 # if HANDLE_MULTIBYTE
301  if (__builtin_expect (MB_CUR_MAX, 1) != 1)
302  {
303  mbstate_t ps;
304  size_t n;
305  const char *p;
306  wchar_t *wpattern_malloc = NULL;
307  wchar_t *wpattern;
308  wchar_t *wstring_malloc = NULL;
309  wchar_t *wstring;
310  size_t alloca_used = 0;
311 
312  /* Convert the strings into wide characters. */
313  memset (&ps, '\0', sizeof (ps));
314  p = pattern;
315 #ifdef _LIBC
316  n = __strnlen (pattern, 1024);
317 #else
318  n = strlen (pattern);
319 #endif
320  if (__glibc_likely (n < 1024))
321  {
322  wpattern = (wchar_t *) alloca_account ((n + 1) * sizeof (wchar_t),
323  alloca_used);
324  n = mbsrtowcs (wpattern, &p, n + 1, &ps);
325  if (__glibc_unlikely (n == (size_t) -1))
326  /* Something wrong.
327  XXX Do we have to set `errno' to something which mbsrtows hasn't
328  already done? */
329  return -1;
330  if (p)
331  {
332  memset (&ps, '\0', sizeof (ps));
333  goto prepare_wpattern;
334  }
335  }
336  else
337  {
338  prepare_wpattern:
339  n = mbsrtowcs (NULL, &pattern, 0, &ps);
340  if (__glibc_unlikely (n == (size_t) -1))
341  /* Something wrong.
342  XXX Do we have to set `errno' to something which mbsrtows hasn't
343  already done? */
344  return -1;
345  if (__glibc_unlikely (n >= (size_t) -1 / sizeof (wchar_t)))
346  {
347  __set_errno (ENOMEM);
348  return -2;
349  }
350  wpattern_malloc = wpattern
351  = (wchar_t *) malloc ((n + 1) * sizeof (wchar_t));
352  assert (mbsinit (&ps));
353  if (wpattern == NULL)
354  return -2;
355  (void) mbsrtowcs (wpattern, &pattern, n + 1, &ps);
356  }
357 
358  assert (mbsinit (&ps));
359 #ifdef _LIBC
360  n = __strnlen (string, 1024);
361 #else
362  n = strlen (string);
363 #endif
364  p = string;
365  if (__glibc_likely (n < 1024))
366  {
367  wstring = (wchar_t *) alloca_account ((n + 1) * sizeof (wchar_t),
368  alloca_used);
369  n = mbsrtowcs (wstring, &p, n + 1, &ps);
370  if (__glibc_unlikely (n == (size_t) -1))
371  {
372  /* Something wrong.
373  XXX Do we have to set `errno' to something which
374  mbsrtows hasn't already done? */
375  free_return:
376  free (wpattern_malloc);
377  return -1;
378  }
379  if (p)
380  {
381  memset (&ps, '\0', sizeof (ps));
382  goto prepare_wstring;
383  }
384  }
385  else
386  {
387  prepare_wstring:
388  n = mbsrtowcs (NULL, &string, 0, &ps);
389  if (__glibc_unlikely (n == (size_t) -1))
390  /* Something wrong.
391  XXX Do we have to set `errno' to something which mbsrtows hasn't
392  already done? */
393  goto free_return;
394  if (__glibc_unlikely (n >= (size_t) -1 / sizeof (wchar_t)))
395  {
396  free (wpattern_malloc);
397  __set_errno (ENOMEM);
398  return -2;
399  }
400 
401  wstring_malloc = wstring
402  = (wchar_t *) malloc ((n + 1) * sizeof (wchar_t));
403  if (wstring == NULL)
404  {
405  free (wpattern_malloc);
406  return -2;
407  }
408  assert (mbsinit (&ps));
409  (void) mbsrtowcs (wstring, &string, n + 1, &ps);
410  }
411 
412  int res = internal_fnwmatch (wpattern, wstring, wstring + n,
413  flags & FNM_PERIOD, flags, NULL,
414  alloca_used);
415 
416  free (wstring_malloc);
417  free (wpattern_malloc);
418 
419  return res;
420  }
421 # endif /* mbstate_t and mbsrtowcs or _LIBC. */
422 
423  return internal_fnmatch (pattern, string, string + strlen (string),
424  flags & FNM_PERIOD, flags, NULL, 0);
425 }
