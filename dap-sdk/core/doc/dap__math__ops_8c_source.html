32 uint128_t dap_uint128_substract(uint128_t a, uint128_t b)
33 {
34 #ifdef DAP_GLOBAL_IS_INT128
35  if (a < b) {
36  log_it(L_WARNING, "Substract result overflow");
37  return 0;
38  }
39  return a - b;
40 #else
41  uint128_t l_ret = {};
42  if (a.u64[0] < b.u64[0] || (a.u64[0] == b.u64[0] && a.u64[1] < b.u64[1])) {
43  log_it(L_WARNING, "Substract result overflow");
44  return l_ret;
45  }
46  l_ret.u64[0] = a.u64[0] - b.u64[0];
47  l_ret.u64[1] = a.u64[1] - b.u64[1];
48  if (a.u64[1] < b.u64[1])
49  l_ret.u64[0]--;
50  return l_ret;
51 #endif
52 }

60 uint128_t dap_uint128_add(uint128_t a, uint128_t b)
61 {
62 #ifdef DAP_GLOBAL_IS_INT128
63  uint128_t l_ret = a + b;
64  if (l_ret < a || l_ret < b) {
65  log_it(L_WARNING, "Sum result overflow");
66  return 0;
67  }
68 #else
69  uint128_t l_ret = {};
70  l_ret.u64[0] = a.u64[0] + b.u64[0];
71  l_ret.u64[1] = a.u64[1] + b.u64[1];
72  if (l_ret.u64[1] < a.u64[1] || l_ret.u64[1] < b.u64[1])
73  l_ret.u64[0]++;
74  if (l_ret.u64[0] < a.u64[0] || l_ret.u64[0] < b.u64[0]) {
75  log_it(L_WARNING, "Sum result overflow");
76  uint128_t l_nul = {};
77  return l_nul;
78  }
79 #endif
80  return l_ret;
81 }
