40 static uint128_t s_strtou128(const char *p, char **endp, int base)
41 {
42  uint128_t v = 0;
43  int digit;
44 
45  if (base == 0) { /* handle octal and hexadecimal syntax */
46  base = 10;
47  if (*p == '0') {
48  base = 8;
49  if ((p[1] == 'x' || p[1] == 'X') && s_strdigit(p[2]) < 16) {
50  p += 2;
51  base = 16;
52  }
53  }
54  }
55  if (base < 2 || base > 36) {
56  errno = EINVAL;
57  } else
58  if ((digit = s_strdigit(*p)) < base) {
59  v = digit;
60  /* convert to unsigned 128 bit with overflow control */
61  while ((digit = s_strdigit(*++p)) < base) {
62  uint128_t v0 = v;
63  v = v * base + digit;
64  if (v < v0) {
65  v = ~(uint128_t)0;
66  errno = ERANGE;
67  }
68  }
69  if (endp) {
70  *endp = (char *)p;
71  }
72  }
73  return v;
74 }
